    def <<(x)
      case x
      when Element
        @result_array << SanitizedString[@result_str] unless @result_str == ""
        @result_array << x
        @result_str = ""
      when String
        @result_str << x
      when NodeArray
        x.array.each do |n|
          n.pre_format1(self)
        end
      when Node
        @formatter.format(x)
      else
        @result_str << x.to_s
      end
      self
    end

    def result
      @result_array << SanitizedString[@result_str] unless @result_str == ""
      case @result_array.size
      when 0
        Null
      when 1
        @result_array[0]
      else
        @result_array
      end
    end
