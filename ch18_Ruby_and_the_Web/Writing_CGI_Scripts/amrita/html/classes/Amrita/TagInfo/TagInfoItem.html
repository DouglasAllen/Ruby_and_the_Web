      def url_attr?(attr_name)
        return false unless @url_attributes
        case attr_name
        when nil 
          return false
        when Symbol
        when String
          attr_name = attr_name.intern
        else
          attr_name = attr_name.to_s.intern
        end
        @url_attributes[attr_name]
      end
